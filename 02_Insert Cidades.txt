-- SC
	-- Chapecó
	insert into  public.cidade (id_estado,nome)
	select uf.id,'Chapecó' 
	from 
		public.uf uf 
		left join public.cidade cid
			on  cid.id_estado = uf.id
			and cid.nome = 'Chapecó'
	where sigla = 'SC'
	and cid.id is null;
