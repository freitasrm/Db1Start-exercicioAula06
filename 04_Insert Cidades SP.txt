-- inserindo São Caetano do Sul
insert into public.cidade (id_estado,nome)
select
	uf.id,'São Caetano do Sul'
from
	public.uf uf
	left join public.cidade cid
		on  cid.id_estado = uf.id
		and cid.nome = 'São Caetano do Sul'
where
	cid.id is null
and uf.sigla = 'SP';
