-- Inserindo 2 endereços
insert into public.endereco (id_cidade,id_pessoa,logradouro,numero,cep,complemento,tipo_endereco)
select
	cid.id,
	pes.id,
	'Avenida Horácio Racanello Filho' logradouro,
	2220 numero,
	87013100 cep,
	'Laboratório de Informática 01' complemento,
	case 
		row_number() over (partition by pes.id) 
		when 1 
			then 'COMERCIAL' 
			else 'RESIDENCIAL'
	end tipo_endereco
from
	public.pessoa pes
	inner join public.uf uf
		on  uf.sigla = 'PR'
	inner join public.cidade cid
		on  cid.id_estado = uf.id
where
	cid.id in
		(
			select id 
			from
				public.cidade
			where id_estado = uf.id
			
			limit 2);
